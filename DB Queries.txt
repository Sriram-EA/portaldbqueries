-- Users table for Logging in. 

create table users(
    psno varchar(100) primary key, 
    firstname varchar(100), 
    lastname varchar(100),  
    dob varchar(30),
    logindate varchar(100), 
    logintime time, 
    logoutdate varchar(100),
    logouttime time, 
    loginflag varchar(100)
); 
select * from users;  

-------------------------------

-- Events table 

create table events( 
    eventid int primary key,
    eventname varchar(300),  
    eventdate varchar(100), 
    starttime time, 
    endtime time,
    organizername varchar(100), 
    accountname varchar(100), 
    eventflag varchar(100)
); 

-- Items table

create table items(
    itemid int primary key, 
    itemname varchar(300), 
    presentername varchar(100),  
    eventid int,
    foreign key(eventid) references events(eventid)
); 

-- Subscribed events Table 

create table subscribedevents( 

psno varchar(100), 
eventid int, 
foreign key(psno) references users(psno), 
foreign key(eventid) references events(eventid)

); 

-- Scores Table 

create table scores( 
    psno varchar(100), 
    score int, 
    eventid int, 
    itemid int,  
    scoredate varchar(100), 
    scoretime time,
    foreign key(psno) references users(psno),
    foreign key(eventid) references events(eventid),  
    foreign key(itemid) references items(itemid)
);

-- Panelist Table 

create table panelist(
    psno varchar(100), 
    eventid int,  
    foreign key(psno) references users(psno),
    foreign key(eventid) references events(eventid)
);

select * from panelist; 

-- Admin Table

create table admintable(
	psno varchar(100), 
    passwords varchar(300),
    foreign key(psno) references users(psno)
);

select * from admintable;
